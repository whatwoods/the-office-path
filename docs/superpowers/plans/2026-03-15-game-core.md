# The Office Path - 游戏核心 实施计划

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the game engine with all hard rules, type system, save/load, and API routes — a fully testable backend that the AI system and frontend can plug into.

**Architecture:** Next.js App Router for API routes, pure TypeScript modules for the game engine (no framework dependency), Vitest for testing. The engine is a pure-function state machine: given a game state and an action, return a new game state. No side effects in the engine itself.

**Tech Stack:** Next.js 15, TypeScript, Vitest, Zustand (for future frontend state)

```typescript
import type { PlayerAttributes } from '@/types/game'

/** Attributes that are clamped 0-100 */
const CLAMPED_ATTRIBUTES: (keyof PlayerAttributes)[] = [
  'health', 'professional', 'communication', 'management',
  'network', 'mood', 'reputation',
]

/** Starting attributes for a new graduate */
export const INITIAL_ATTRIBUTES: PlayerAttributes = {
  health: 90,
  professional: 15,
  communication: 20,
  management: 5,
  network: 5,
  mood: 70,
  money: 5000,
  reputation: 0,
}

/** Clamp an attribute value to its valid range */
export function clampAttribute(attr: keyof PlayerAttributes, value: number): number {
  if (!CLAMPED_ATTRIBUTES.includes(attr)) return value // money etc. unclamped
  return Math.max(0, Math.min(100, value))
}

/** Apply stat changes to attributes, returns new object (immutable) */
export function applyStatChanges(
  attrs: PlayerAttributes,
  changes: Partial<PlayerAttributes>,
): PlayerAttributes {
  const result = { ...attrs }
  for (const [key, delta] of Object.entries(changes)) {
    const attr = key as keyof PlayerAttributes
    if (attr in result && typeof delta === 'number') {
      result[attr] = clampAttribute(attr, result[attr] + delta)
    }
  }
  return result
}

  owned: 10,

  attrs: PlayerAttributes,
  housingType: string,

 * Design choice: worst debuff takes precedence (no stacking).
 * Subhealth (health≤30) = 0.5, Low mood (mood≤30) = 0.7 */
export function getEffectMultiplier(attrs: PlayerAttributes): number {

  if (hasLowMood(attrs)) return 0.7    // 低心情: -30%
  return 1.0
```

```typescript
import type { ActionAllocation, QuarterPlan, GameAction } from '@/types/actions'
import { ACTION_STAMINA_COST } from '@/types/actions'

/** Result of processing a single action */
export interface ActionResult {
  statChanges: Partial<PlayerAttributes>
  projectProgress: number       // 0 or 1
  workActionCount: number       // 0 or 1
  npcFavorTarget?: string       // NPC id for favor change
  npcFavorAmount?: number       // favor delta
  sideHustleIncome?: number     // random income from side hustle
  sideHustleCaught?: boolean    // whether caught doing side hustle
}

/** Process a single action and return its effects */
export function processAction(
  attrs: PlayerAttributes,
  allocation: ActionAllocation,
  effectMultiplier: number,
  randomFn?: () => number, // for deterministic testing
): ActionResult {
  const m = effectMultiplier
  const result: ActionResult = {
    statChanges: {},
    projectProgress: 0,
    workActionCount: 0,
  }

  switch (allocation.action) {
    case 'work_hard':
      result.statChanges = {
        professional: Math.floor(3 * m),
        health: -2,  // health cost not reduced by multiplier
      }
      result.projectProgress = 1
      result.workActionCount = 1
      break

    case 'study':
      if (allocation.target) {
        result.statChanges = { [allocation.target]: Math.floor(5 * m) }
      }
      break

    case 'socialize':
      result.statChanges = {
        communication: Math.floor(2 * m),
        network: Math.floor(3 * m),
        money: -500,
      }
      if (allocation.target) {
        // Favor gain: 5-15 based on communication skill
        const favorGain = Math.floor((5 + (attrs.communication / 10)) * m)
        result.npcFavorTarget = allocation.target
        result.npcFavorAmount = favorGain
      }
      break

    case 'manage_up':
      result.statChanges = { communication: Math.floor(1 * m) }
      // Leader favor +10, random coworker favor -5
      // These are handled by the caller since they need NPC context
      result.npcFavorTarget = '__leader__'
      result.npcFavorAmount = 10
      break

    case 'slack_off':
      result.statChanges = {
        mood: Math.floor(8 * m),
        health: 3,
      }
      break

    case 'side_hustle': {
      const rng = randomFn ?? Math.random
      const income = 2000 + Math.floor(rng() * 6001) // 2000-8000
      const caught = rng() < 0.2
      result.sideHustleIncome = income
      result.sideHustleCaught = caught
      // Spec: income is always earned, caught adds penalties on top
      result.statChanges = { money: income }
      if (caught) {
        result.statChanges.reputation = -10
        // Leader favor -20 handled by caller
      }
      break
    }

    case 'job_interview':
      result.statChanges = { network: Math.floor(2 * m) }
      // Offer generation handled by agent layer
      break

    case 'resign_startup':
      // No stat changes, triggers phase transition
      break

    // Phase 2 actions
    case 'improve_product':
      result.statChanges = {} // company attribute, not player
      break
    case 'recruit':
    case 'fundraise':
    case 'team_manage':
    case 'biz_develop':
    case 'marketing':
      result.statChanges = {}
      break
    case 'rest':
      result.statChanges = {
        health: 3,
        mood: Math.floor(8 * m),
      }
      break
  }

  return result
}

/** Validate a quarter plan against stamina budget */
export function validateQuarterPlan(
  plan: QuarterPlan,
  maxStamina: number,
): { valid: boolean; totalCost: number; error?: string } {
  const totalCost = getTotalStaminaCost(plan)
  if (totalCost > maxStamina) {
    return {
      valid: false,
      totalCost,
      error: `Total stamina cost ${totalCost} exceeds budget ${maxStamina}`,
    }
  }
  return { valid: true, totalCost }
}

/** Calculate total stamina cost of a plan */
export function getTotalStaminaCost(plan: QuarterPlan): number {
  return plan.actions.reduce(
    (sum, a) => sum + (ACTION_STAMINA_COST[a.action] ?? 0),
    0,
  )
}
```

```typescript
import type { PerformanceRating } from '@/types/game'

/**
 * Calculate performance rating.
 * Score = (workActionCount * 10) + (professional / 2) + random(-10, +10)
 */
export function calculatePerformance(
  workActionCount: number,
  professionalAbility: number,
  randomOffset?: number, // for testing; if undefined, generate random
): PerformanceRating {
  const random = randomOffset ?? (Math.random() * 20 - 10)
  const score = (workActionCount * 10) + (professionalAbility / 2) + random

  return scoreToRating(score)
}

/** Convert numeric score to rating */
export function scoreToRating(score: number): PerformanceRating {
  if (score >= 80) return 'S'
  if (score >= 60) return 'A'
  if (score >= 45) return 'B+'
  if (score >= 25) return 'B'
  return 'C'
}

/** Convert rating to a comparable number for sorting */
export function ratingToScore(rating: PerformanceRating): number {
  switch (rating) {
    case 'S': return 5
    case 'A': return 4
    case 'B+': return 3
    case 'B': return 2
    case 'C': return 1
  }
}

/** Ratings that satisfy "≥B+" condition */
const PASSING_RATINGS: PerformanceRating[] = ['S', 'A', 'B+']

/**
 * Check if the last N performance ratings all meet ≥B+.
 * Used for promotion conditions.
 */
export function meetsPromotionPerformance(
  history: PerformanceRating[],
  consecutiveCount: number,
): boolean {
  if (history.length < consecutiveCount) return false
  const lastN = history.slice(-consecutiveCount)
  return lastN.every((r) => PASSING_RATINGS.includes(r))
}

/**
 * Check L3→L4 condition: "最近2次考核中≥1次A/S或≥2次B+"
 */
export function meetsL3ToL4Performance(history: PerformanceRating[]): boolean {
  if (history.length < 1) return false
  const last2 = history.slice(-2)
  const hasAS = last2.some((r) => r === 'A' || r === 'S')
  const bPlusCount = last2.filter((r) => r === 'B+').length
  return hasAS || bPlusCount >= 2
}

/**
 * Apply performance rating effects: salary raise + reputation change.
 * S: +15% salary, +5 reputation
 * A: +10% salary, +2 reputation
 * B+: +5% salary
 * B: no change
 * C: warning (consecutive 2 C's triggers layoff, handled by event agent)
 */
export interface PerformanceEffects {
  salaryMultiplier: number  // e.g., 1.15 for 15% raise
  reputationChange: number
  warning: boolean
}

export function getPerformanceEffects(rating: PerformanceRating): PerformanceEffects {
  switch (rating) {
    case 'S':  return { salaryMultiplier: 1.15, reputationChange: 5, warning: false }
    case 'A':  return { salaryMultiplier: 1.10, reputationChange: 2, warning: false }
    case 'B+': return { salaryMultiplier: 1.05, reputationChange: 0, warning: false }
    case 'B':  return { salaryMultiplier: 1.00, reputationChange: 0, warning: false }
    case 'C':  return { salaryMultiplier: 1.00, reputationChange: 0, warning: true }
  }
}
```

```typescript
import type { GameState, JobLevel } from '@/types/game'
import { meetsPromotionPerformance, meetsL3ToL4Performance } from './performance'

export interface PromotionResult {
  eligible: boolean
  nextLevels: JobLevel[]
  failReasons: string[]
}

/** Get possible next levels from current level */
export function getNextLevels(current: JobLevel): JobLevel[] {
  const map: Record<JobLevel, JobLevel[]> = {
    L1: ['L2'],
    L2: ['L3'],
    L3: ['L4'],
    L4: ['L5'],
    L5: ['L6_tech', 'L6_mgmt'],
    L6_tech: ['L7_tech'],
    L6_mgmt: ['L7_mgmt'],
    L7_tech: ['L8'],
    L7_mgmt: ['L8'],
    L8: [],
  }
  return map[current] ?? []
}

/** Check if the player is eligible for promotion */
export function checkPromotion(state: GameState): PromotionResult {
  const { player, job, projectProgress, performanceWindow } = state
  const failReasons: string[] = []
  const eligibleLevels: JobLevel[] = []

  const nextLevels = getNextLevels(job.level)
  if (nextLevels.length === 0) {
    return { eligible: false, nextLevels: [], failReasons: ['Already at max level'] }
  }

  for (const target of nextLevels) {
    const reasons = checkLevelConditions(state, target)
    if (reasons.length === 0) {
      eligibleLevels.push(target)
    } else {
      failReasons.push(...reasons.map((r) => `${target}: ${r}`))
    }
  }

  return {
    eligible: eligibleLevels.length > 0,
    nextLevels: eligibleLevels,
    failReasons,
  }
}

/** Get the leader's favor toward the player */
function getLeaderFavor(state: GameState): number {
  const leader = state.npcs.find(
    (n) => n.role === '直属领导' && n.isActive
  )
  return leader?.favor ?? 0
}

/** Check conditions for a specific target level */
function checkLevelConditions(state: GameState, target: JobLevel): string[] {
  const { player, job, projectProgress, performanceWindow } = state
  const history = performanceWindow.history
  const reasons: string[] = []

  switch (target) {

      if (player.network < 30) reasons.push('人脉不足30')
      if (getLeaderFavor(state) < 60) reasons.push('直属领导好感不足60')
      break

    case 'L7_tech': // L6_tech→L7
      if (player.professional < 80) reasons.push('专业能力不足80')
      if (player.reputation < 50) reasons.push('声望不足50')
      if (!meetsPromotionPerformance(history, 3)) reasons.push('绩效不达标(需连续3次≥B+)')

      if (highAbilities < 2) reasons.push('不足2项能力达到80')
      // Opportunity event is checked separately (not a pure stat check)
      break
  }

  return reasons
```

```typescript
import type { HousingType } from '@/types/game'
import { RENT_TABLE } from '@/types/game'

/** Base quarterly expenses (excluding rent) */
const BASE_FOOD = 3000
const BASE_TRANSPORT = 1500
const SLUM_TRANSPORT_EXTRA = 500
const BASE_DAILY = 1500

export interface QuarterlyExpenses {
  rent: number
  food: number
  transport: number
  daily: number
  total: number
}

export interface QuarterlyEconomyResult {
  income: number
  expenses: QuarterlyExpenses
  net: number
}

/** Calculate quarterly income from salary */
export function calculateQuarterlyIncome(monthlySalary: number): number {
  return monthlySalary * 3
}

/** Calculate quarterly expenses based on housing */
export function calculateQuarterlyExpenses(housing: HousingType): QuarterlyExpenses {
  const rent = RENT_TABLE[housing] * 3
  const food = BASE_FOOD
  const transport = BASE_TRANSPORT + (housing === 'slum' ? SLUM_TRANSPORT_EXTRA : 0)
  const daily = BASE_DAILY
  return {
    rent,
    food,
    transport,
    daily,
    total: rent + food + transport + daily,
  }
}

/** Calculate full quarterly economy result */
export function applyQuarterlyEconomy(
  monthlySalary: number,
  housing: HousingType,
): QuarterlyEconomyResult {
  const income = calculateQuarterlyIncome(monthlySalary)
  const expenses = calculateQuarterlyExpenses(housing)
  return {
    income,
    expenses,
    net: income - expenses.total,
  }
}

/** Apply salary raise by percentage */
export function applySalaryRaise(currentSalary: number, raisePercent: number): number {
  return Math.round(currentSalary * (1 + raisePercent / 100))
}

/** Check if money is negative (月光族 status) */
export function isBroke(money: number): boolean {
  return money < 0
}

/** Down payment required for buying a house */
export const DOWN_PAYMENT = 600000
```

```typescript
import type { CompanyState, CompanyStage } from '@/types/company'
import { STAGE_BASE_REVENUE, STAGE_BASE_CUSTOMERS, OFFICE_RENT } from '@/types/company'
import type { EconomyStatus } from '@/types/agents'
import { ECONOMY_COEFFICIENT } from '@/types/agents'
import type { Phase2Action } from '@/types/actions'

/** Average employee salary by company stage (monthly, in yuan) */
const STAGE_AVG_SALARY: Record<CompanyStage, number> = {
  garage: 8000,
  small_team: 12000,
  series_a: 18000,
  growth: 25000,
  pre_ipo: 35000,
  public: 45000,
}

/** Calculate quarterly revenue */
export function calculateQuarterlyRevenue(
  company: CompanyState,
  economy: EconomyStatus,
): number {
  const baseRevenue = STAGE_BASE_REVENUE[company.stage]
  const qualityCoeff = company.productQuality / 100
  const marketCoeff = ECONOMY_COEFFICIENT[economy]
  const baseCustomers = STAGE_BASE_CUSTOMERS[company.stage]
  const customerScale = Math.min(company.customerCount / baseCustomers, 2.0)
  const brandBonus = company.brandAwareness / 200

  return Math.round(
    baseRevenue * qualityCoeff * marketCoeff * customerScale * (1 + brandBonus)
  )
}

/** Calculate quarterly company expenses */
export function calculateQuarterlyCompanyExpenses(
  company: CompanyState,
  founderMonthlySalary: number,
): number {
  const avgSalary = STAGE_AVG_SALARY[company.stage]
  const wages = company.employeeCount * avgSalary * 3
  const founderWages = founderMonthlySalary * 3
  const rent = OFFICE_RENT[company.officeType]
  const operations = company.employeeCount * 2000
  return wages + founderWages + rent + operations
}

/** Calculate company valuation */
export function calculateValuation(
  quarterlyRevenue: number,
  peMultiple: number, // 5-30, from World Agent
): number {
  const annualRevenue = quarterlyRevenue * 4
  return Math.round(annualRevenue * peMultiple)
}

/** Process a Phase 2 action on company state (immutable) */
export function processCompanyAction(
  company: CompanyState,
  action: Phase2Action,
  effectMultiplier: number,
): CompanyState {
  const result = { ...company }
  const m = effectMultiplier

  switch (action) {
    case 'improve_product':
      result.productQuality = Math.min(100, result.productQuality + Math.floor(5 * m))
      break
    case 'recruit':
      result.employeeCount += Math.floor((1 + Math.random() * 2) * m) // 1-3
      break
    case 'team_manage':
      result.teamSatisfaction = Math.min(100, result.teamSatisfaction + Math.floor(8 * m))
      break
    case 'biz_develop':
      result.customerCount += Math.floor((2 + Math.random() * 3) * m) // 2-5
      break
    case 'marketing':
      result.brandAwareness = Math.min(100, result.brandAwareness + Math.floor(5 * m))
      break
    case 'fundraise':
    case 'rest':
      // handled elsewhere
      break
  }

  return result
}

/** Check company upgrade conditions (spec section 5) */
export function checkCompanyUpgrade(company: CompanyState): boolean {
  switch (company.stage) {
    case 'garage':
      // 季度营收>=10万, 连续2季度盈利
      return company.quarterlyRevenue >= 100000
        && company.consecutiveProfitableQuarters >= 2
    case 'small_team':
      return company.quarterlyRevenue >= 500000 && company.teamSatisfaction >= 60
    case 'series_a':
      // 季度营收>=200万, 获得A轮融资
      return company.quarterlyRevenue >= 2000000 && company.hasSeriesAFunding === true
    case 'growth':
      // 季度营收>=1000万, 年增长率>=50%
      return company.quarterlyRevenue >= 10000000 && company.annualGrowthRate >= 0.5
    case 'pre_ipo':
      return false // requires IPO critical period
    case 'public':
      return false // terminal
  }
}

/** Team satisfaction decays each quarter */
export function applyTeamSatisfactionDecay(company: CompanyState): CompanyState {
  return {
    ...company,
    teamSatisfaction: Math.max(0, company.teamSatisfaction - 3),
  }
}
```

```typescript
import type { CriticalPeriod, CriticalPeriodType, TimeMode, HousingType } from '@/types/game'

/** Critical period configuration */
export const CRITICAL_PERIOD_CONFIG: Record<CriticalPeriodType, { maxDays: number }> = {
  onboarding: { maxDays: 5 },
  promotion_review: { maxDays: 3 },
  company_crisis: { maxDays: 7 }, // 5-10, use 7 as default
  project_sprint: { maxDays: 5 }, // 3-7, use 5 as default
  job_negotiation: { maxDays: 3 },
  startup_launch: { maxDays: 7 },
  fundraising: { maxDays: 5 },
  ipo_review: { maxDays: 10 },
}

/** Advance to next quarter */
export function advanceQuarter(currentQuarter: number): number {
  return currentQuarter + 1
}

/** Get max stamina based on mode and housing */
export function getMaxStamina(mode: TimeMode, housingType: HousingType): number {
  if (mode === 'critical') return 3
  // Quarterly mode: 10 base, slum housing -1
  return housingType === 'slum' ? 9 : 10
}

/** Enter a critical period */
export function enterCriticalPeriod(type: CriticalPeriodType): CriticalPeriod {
  const config = CRITICAL_PERIOD_CONFIG[type]
  return {
    type,
    currentDay: 1,
    maxDays: config.maxDays,
    staminaPerDay: 3,
  }
}

/** Advance one day in a critical period */
export function advanceCriticalDay(
  period: CriticalPeriod,
): { period: CriticalPeriod; isComplete: boolean } {
  const newDay = period.currentDay + 1
  const isComplete = newDay > period.maxDays
  return {
    period: { ...period, currentDay: isComplete ? period.maxDays : newDay },
    isComplete,
  }
}

/** Check if it's time for performance review (every 2 quarters) */
export function isPerformanceReviewQuarter(quarter: number): boolean {
  return quarter > 0 && quarter % 2 === 0
}
```

```typescript
/** Migration function type */
type MigrationFn = (data: Record<string, unknown>) => Record<string, unknown>

/**
 * Registry of migrations. Key is "fromVersion→toVersion".
 * Add entries here as the save format evolves.
 */
const MIGRATIONS: Record<string, MigrationFn> = {
  // Example for future:
  // '1.0→1.1': (data) => ({ ...data, version: '1.1', newField: 'default' }),
}

/** Ordered list of versions */
const VERSION_CHAIN = ['1.0']

/**
 * Migrate save data to target version.
 * Returns migrated data or null if migration is impossible.
 */
export function migrate(
  data: Record<string, unknown>,
  targetVersion: string,
): Record<string, unknown> | null {
  const currentVersion = data.version as string

  if (currentVersion === targetVersion) return data

  // Find position in version chain
  const fromIndex = VERSION_CHAIN.indexOf(currentVersion)
  const toIndex = VERSION_CHAIN.indexOf(targetVersion)

  if (fromIndex === -1 || toIndex === -1 || fromIndex >= toIndex) {
    return null // unknown version or downgrade not supported
  }

  // Apply migrations in sequence
  let result = { ...data }
  for (let i = fromIndex; i < toIndex; i++) {
    const key = `${VERSION_CHAIN[i]}→${VERSION_CHAIN[i + 1]}`
    const fn = MIGRATIONS[key]
    if (!fn) return null // missing migration step
    result = fn(result)
  }

  return result
}
```

```typescript
import type { GameState } from '@/types/game'
import { migrate } from './migration'

const CURRENT_VERSION = '1.0'
const STORAGE_PREFIX = 'office_path_save_'

/** Available save slots */
export const SAVE_SLOTS = ['auto', 'slot1', 'slot2', 'slot3'] as const
export type SaveSlot = (typeof SAVE_SLOTS)[number]

/** Save metadata for display */
export interface SaveMeta {
  slot: SaveSlot
  quarter: number
  phase: number
  level: string
  savedAt: string
}

/** Save game state to a slot */
export function saveGame(state: GameState, slot: SaveSlot): void {
  const payload = {
    state,
    savedAt: new Date().toISOString(),
  }
  localStorage.setItem(
    STORAGE_PREFIX + slot,
    JSON.stringify(payload),
  )
}

/** Load game state from a slot */
export function loadGame(slot: SaveSlot): GameState | null {
  const raw = localStorage.getItem(STORAGE_PREFIX + slot)
  if (!raw) return null

  try {
    const payload = JSON.parse(raw)
    const state = payload.state as GameState

    // Check version and migrate if needed
    if (state.version !== CURRENT_VERSION) {
      const migrated = migrate(
        state as unknown as Record<string, unknown>,
        CURRENT_VERSION,
      )
      if (!migrated) return null // migration failed
      return migrated as unknown as GameState
    }

    return state
  } catch {
    return null
  }
}

/** List all saves with metadata */
export function listSaves(): SaveMeta[] {
  const saves: SaveMeta[] = []
  for (const slot of SAVE_SLOTS) {
    const raw = localStorage.getItem(STORAGE_PREFIX + slot)
    if (!raw) continue
    try {
      const payload = JSON.parse(raw)
      const state = payload.state as GameState
      saves.push({
        slot,
        quarter: state.currentQuarter,
        phase: state.phase,
        level: state.job.level,
        savedAt: payload.savedAt,
      })
    } catch {
      // skip corrupted saves
    }
  }
  return saves
}

/** Delete a save slot */
export function deleteSave(slot: SaveSlot): void {
  localStorage.removeItem(STORAGE_PREFIX + slot)
}
```

```typescript
import type { GameState, PerformanceRating } from '@/types/game'
import type { QuarterPlan } from '@/types/actions'
import { processAction } from './actions'
import { validateQuarterPlan } from './actions'
import { applyStatChanges, applyQuarterlyHealthDecay, applyQuarterlyHousingMood, getEffectMultiplier } from './attributes'
import { applyQuarterlyEconomy, isBroke } from './economy'
import { calculatePerformance, getPerformanceEffects } from './performance'
import { advanceQuarter, getMaxStamina, isPerformanceReviewQuarter } from './time'

export interface QuarterResult {
  state: GameState
  performanceRating?: PerformanceRating
  salaryChange?: number
}

/**
 * Settle a full quarter: apply all player actions, economy, health decay,
 * and optionally run performance review.
 */
export function settleQuarter(
  state: GameState,
  plan: QuarterPlan,
): QuarterResult {
  const maxStamina = getMaxStamina('quarterly', state.housing.type)
  const validation = validateQuarterPlan(plan, maxStamina)
  if (!validation.valid) {
    throw new Error(validation.error ?? 'Invalid plan: exceeds stamina budget')
  }

  let newState: GameState = JSON.parse(JSON.stringify(state)) // deep clone
  const effectMultiplier = getEffectMultiplier(newState.player)

  // 1. Process each action
  let workCount = 0
  for (const allocation of plan.actions) {
    const result = processAction(newState.player, allocation, effectMultiplier)

    // Apply stat changes
    if (Object.keys(result.statChanges).length > 0) {
      newState.player = applyStatChanges(newState.player, result.statChanges)
    }

    // Track project progress
    newState.projectProgress.currentProgress += result.projectProgress
    if (newState.projectProgress.currentProgress >= 3) {
      newState.projectProgress.completed += 1
      newState.projectProgress.currentProgress -= 3
    }

    // Track work actions for performance review
    workCount += result.workActionCount

    // Apply NPC favor changes
    if (result.npcFavorTarget && result.npcFavorAmount) {
      const npc = newState.npcs.find((n) =>
        result.npcFavorTarget === '__leader__'
          ? n.role === '直属领导' && n.isActive
          : n.id === result.npcFavorTarget
      )
      if (npc) {
        npc.favor = Math.max(0, Math.min(100, npc.favor + result.npcFavorAmount))
      }
    }
  }

  // 2. Apply quarterly health decay
  newState.player = applyQuarterlyHealthDecay(newState.player, newState.housing.type)

  // 2b. Apply housing mood effect (slum -5, studio +3, apartment +5, luxury +8, owned +10)
  newState.player = applyQuarterlyHousingMood(newState.player, newState.housing.type)

  // 2c. Apply housing reputation bonus (luxury +2, owned +5)
  const housingRepBonus: Record<string, number> = { luxury: 2, owned: 5 }
  const repBonus = housingRepBonus[newState.housing.type] ?? 0
  if (repBonus > 0) {
    newState.player = applyStatChanges(newState.player, { reputation: repBonus })
  }

  // 3. Apply economy
  const econ = applyQuarterlyEconomy(newState.job.salary, newState.housing.type)
  newState.player = applyStatChanges(newState.player, { money: econ.net })

  // 4. Apply broke penalty
  if (isBroke(newState.player.money)) {
    newState.player = applyStatChanges(newState.player, { mood: -10 })
  }

  // 5. Update performance window
  newState.performanceWindow.workActionCount += workCount
  newState.performanceWindow.quartersInWindow += 1

  // 6. Advance quarter
  newState.currentQuarter = advanceQuarter(newState.currentQuarter)
  newState.job.quartersAtLevel += 1
  newState.job.totalQuarters += 1

  // 7. Performance review if applicable
  let performanceRating: PerformanceRating | undefined
  let salaryChange: number | undefined
  if (isPerformanceReviewQuarter(newState.currentQuarter)) {
    performanceRating = calculatePerformance(
      newState.performanceWindow.workActionCount,
      newState.player.professional,
    )
    newState.performanceWindow.history.push(performanceRating)

    // Apply salary raise and reputation change via getPerformanceEffects
    const effects = getPerformanceEffects(performanceRating)
    if (effects.salaryMultiplier > 1) {
      const oldSalary = newState.job.salary
      newState.job.salary = Math.round(oldSalary * effects.salaryMultiplier)
      salaryChange = newState.job.salary - oldSalary
    }
    if (effects.reputationChange > 0) {
      newState.player = applyStatChanges(newState.player, { reputation: effects.reputationChange })
    }

    // Reset performance window
    newState.performanceWindow.workActionCount = 0
    newState.performanceWindow.quartersInWindow = 0
  }

  // 8. Reset stamina
  newState.staminaRemaining = getMaxStamina('quarterly', newState.housing.type)

  return { state: newState, performanceRating, salaryChange }
}
```
