# AI Agent System Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the 4-agent AI pipeline (World → Event → NPC → Narrative) with orchestration, conflict resolution, and structured output via Vercel AI SDK v6.

**Architecture:** Each agent is a standalone module with its own Zod schema, system prompt, and `generateText()` call using the `Output.object()` API. An orchestration layer coordinates the sequential pipeline for both quarterly settlement and critical-period daily flows. A conflict resolution module validates agent outputs against game engine hard rules before they reach the narrative layer.

**Tech Stack:** Vercel AI SDK v6 (`ai`), `@ai-sdk/openai`, `@ai-sdk/anthropic`, `@ai-sdk/deepseek`, `zod`, existing game engine from Plan 1.

// src/ai/provider.ts
import { createOpenAI } from '@ai-sdk/openai'
import { createAnthropic } from '@ai-sdk/anthropic'
import { createDeepSeek } from '@ai-sdk/deepseek'

/**
 * Model spec format: "provider:model-id"
 * Examples: "openai:gpt-4o", "anthropic:claude-sonnet-4-20250514", "deepseek:deepseek-chat"
 */
export type ModelSpec = `${'openai' | 'anthropic' | 'deepseek'}:${string}`

const providers = {
  openai: createOpenAI({ apiKey: process.env.OPENAI_API_KEY ?? '' }),
  anthropic: createAnthropic({ apiKey: process.env.ANTHROPIC_API_KEY ?? '' }),
  deepseek: createDeepSeek({ apiKey: process.env.DEEPSEEK_API_KEY ?? '' }),
} as const

type ProviderName = keyof typeof providers

export function getModel(spec: ModelSpec) {
  if (!spec.includes(':')) {
    throw new Error(`Invalid model spec: "${spec}". Expected format "provider:model-id"`)
  }

  const [providerName, ...rest] = spec.split(':')
  const modelId = rest.join(':')

  if (!(providerName in providers)) {
    throw new Error(`Unknown AI provider: ${providerName}`)
  }

  const provider = providers[providerName as ProviderName]
  return provider(modelId)
}

/** Default model specs per agent role. Overridable via env vars. */
export const AGENT_MODELS = {
  world: (process.env.WORLD_AGENT_MODEL ?? 'openai:gpt-4o-mini') as ModelSpec,
  event: (process.env.EVENT_AGENT_MODEL ?? 'openai:gpt-4o-mini') as ModelSpec,
  npc: (process.env.NPC_AGENT_MODEL ?? 'openai:gpt-4o') as ModelSpec,
  narrative: (process.env.NARRATIVE_AGENT_MODEL ?? 'openai:gpt-4o') as ModelSpec,
} as const
```

function buildSystemPrompt(
  input: AgentInput,
  isCriticalPeriod: boolean,
  generateChoices: boolean = isCriticalPeriod,
): string {
  let prompt = `你是"打工之道"游戏的故事编剧（Narrative Agent）。
你负责把所有发生的事情写成好看的故事。

文风要求：
- 幽默、接地气的打工人口吻，偶尔扎心
- 有画面感，像读小说一样
- 用第二人称"你"叙述
- 语言生动，避免流水账
- 300-500字为宜

当前状态：
- 游戏阶段：第${input.state.phase}阶段
- 当前季度：第${input.state.currentQuarter}季度
- 玩家职级：${input.state.job.level}（${input.state.job.companyName}）`

  if (isCriticalPeriod && input.state.criticalPeriod) {
    const cp = input.state.criticalPeriod
    const categories = CRITICAL_PERIOD_CATEGORIES[cp.type] ?? ['行动']
    prompt += `

当前处于关键期：${cp.type}（第${cp.currentDay}天/共${cp.maxDays}天）
每天体力：${cp.staminaPerDay}点，玩家剩余体力：${input.state.staminaRemaining}点

你需要：
1. 生成当天的叙事（150-300字）`

    if (generateChoices) {
      prompt += `
2. 生成3-4个行动选项（choices），每个选项必须包含：
   - choiceId: 唯一标识（格式：${cp.type}_d${cp.currentDay}_a/b/c）
   - label: 选项显示文本
   - staminaCost: 体力消耗（1-2点，不能超过剩余体力${input.state.staminaRemaining}点）
   - effects: 结构化效果（statChanges, npcFavorChanges, riskEvent）
   - category: 必须是以下之一：${categories.join('、')}

选项设计原则：
- 每个选项有不同的策略倾向和风险收益权衡
- 至少一个低风险选项、一个高收益高风险选项
- 效果数值要合理（属性变化±1~5，好感变化±5~15）`
    } else {
      prompt += `
2. 今天是该关键期的最后一天，只生成当天叙事，不要生成choices。`
    }
  } else {
    prompt += `

你需要：
1. 生成季度叙事总结（300-500字），将玩家行动、事件和NPC反应编织成一段连贯的故事
2. 生成narrativeSummary：一句话总结本季度核心事件（用于历史记录）
不需要生成choices（季度模式没有选择卡片）。`
  }

  return prompt
}

function buildUserPrompt(
  input: AgentInput,
  worldContext: WorldAgentOutput,
  eventContext: EventAgentOutput,
  npcContext: NPCAgentOutput,
  playerActions: ActionAllocation[],
  playerContext?: string,
): string {
  const actions = playerActions
    .map(a => a.target ? `${a.action}（目标：${a.target}）` : a.action)
    .join(', ')

  const events = eventContext.events
    .map(e => `【${e.title}】${e.description}`)
    .join('\n')

  const npcActions = npcContext.npcActions
    .map(a => `${a.npcName}${a.action}${a.dialogue ? `，说："${a.dialogue}"` : ''}`)
    .join('\n')

  const history = input.recentHistory
    .map(h => `Q${h.quarter}: ${h.narrativeSummary}`)
    .join('\n')

  return `世界环境：经济${worldContext.economy}，公司${worldContext.companyStatus}
行业趋势：${worldContext.trends.join('、') || '无'}

玩家行动：${actions || '无'}

本季度事件：
${events || '无'}

NPC动态：
${npcActions || '无'}

${history ? `最近历史：\n${history}` : ''}
${playerContext ? `\n关键期玩家选择：${playerContext}` : ''}

请编织以上素材，生成本${input.state.timeMode === 'critical' ? '天' : '季度'}的叙事。`
}

export async function runNarrativeAgent(
  input: AgentInput,
  worldContext: WorldAgentOutput,
  eventContext: EventAgentOutput,
  npcContext: NPCAgentOutput,
  playerActions: ActionAllocation[],
  isCriticalPeriod: boolean,
  playerContext?: string,
  generateChoices: boolean = isCriticalPeriod,
): Promise<NarrativeAgentOutput> {
  const { output } = await generateText({
    model: getModel(AGENT_MODELS.narrative),
    output: Output.object({ schema: NarrativeAgentOutputSchema }),
    system: buildSystemPrompt(input, isCriticalPeriod, generateChoices),
    prompt: buildUserPrompt(input, worldContext, eventContext, npcContext, playerActions, playerContext),
  })

  return output!
}

// src/ai/orchestration/history.ts
import type { PlayerAttributes, QuarterSummary } from '@/types/game'

const MAX_RECENT_HISTORY = 3

/** Return the last N quarter summaries for agent context (spec: max 3). */
export function getRecentHistory(history: QuarterSummary[]): QuarterSummary[] {
  if (history.length <= MAX_RECENT_HISTORY) {
    return history
  }
  return history.slice(-MAX_RECENT_HISTORY)
}

/** Create a structured quarter summary for the history log. */
export function createQuarterSummary(
  quarter: number,
  before: PlayerAttributes,
  after: PlayerAttributes,
  keyEvents: string[],
  npcChanges: string[],
  narrativeSummary: string,
): QuarterSummary {
  const statChanges: Partial<PlayerAttributes> = {}
  for (const key of Object.keys(before) as Array<keyof PlayerAttributes>) {
    const diff = after[key] - before[key]
    if (diff !== 0) {
      statChanges[key] = diff
    }
  }

  return {
    quarter,
    keyEvents,
    statChanges,
    npcChanges,
    narrativeSummary,
  }
}

const BOOM_INCOMPATIBLE_KEYWORDS = ['裁员', '倒闭', '破产', '大萧条', '寒冬']

/**
 * Validate events against world context.
 * Rule: "事件 Agent 不能在行业繁荣时生成全行业大裁员"
 */
export function validateEvents(
  events: GameEvent[],
  worldContext: WorldAgentOutput,
): GameEvent[] {
  if (worldContext.economy !== 'boom') {
    return events
  }

  return events.filter(event => {
    if (event.type !== 'industry' && event.severity !== 'critical') {
      return true
    }
    const text = `${event.title}${event.description}`
    return !BOOM_INCOMPATIBLE_KEYWORDS.some(kw => text.includes(kw))
  })
}

/**
 * NPC consistency check per spec section 九:
 * "编排层检查事件是否与 NPC 性格严重矛盾。若矛盾，该事件被丢弃。
 *  最多丢弃 2 个事件/季度，防止无限丢弃导致空季度"
 *
 * Heuristic: If an event's title/description mentions an NPC by name,
 * and the described action contradicts their personality keywords, discard it.
 * This is a keyword-based check — future improvement could use AI for deeper analysis.
 */
const PERSONALITY_CONTRADICTION_MAP: Record<string, string[]> = {
  '和善': ['暴怒', '大骂', '破口大骂', '发飙', '暴打'],
  '内向': ['公开演讲', '高调', '大声宣布'],
  '精于算计': ['冲动', '鲁莽'],
  '热心': ['冷漠', '拒绝帮忙', '袖手旁观'],
}

export function validateEventNPCConsistency(
  events: GameEvent[],
  npcs: NPC[],
): GameEvent[] {
  let discardCount = 0

  return events.filter(event => {
    if (discardCount >= MAX_DISCARDS_PER_QUARTER) return true

    const text = `${event.title}${event.description}`

    for (const npc of npcs) {
      if (!npc.isActive || !text.includes(npc.name)) continue

      // Check if event contradicts NPC personality
      for (const [trait, contradictions] of Object.entries(PERSONALITY_CONTRADICTION_MAP)) {
        if (npc.personality.includes(trait) && contradictions.some(c => text.includes(c))) {
          discardCount++
          return false
        }
      }
    }

    return true
  })
}

/**
 * Validate NPC actions for consistency.
 * Rules:
 * - Filter actions for non-existent/inactive NPCs
 * - Cap favor changes to ±MAX_FAVOR_CHANGE_PER_ACTION
 * - Max MAX_DISCARDS_PER_QUARTER discards to prevent empty quarters
 */
export function validateNPCActions(
  npcOutput: NPCAgentOutput,
  activeNpcs: NPC[],
): NPCAgentOutput {
  const activeNames = new Set(activeNpcs.filter(n => n.isActive).map(n => n.name))
  let discardCount = 0

  const validActions = npcOutput.npcActions.filter(action => {
    if (activeNames.has(action.npcName)) {
      return true
    }
    if (discardCount < MAX_DISCARDS_PER_QUARTER) {
      discardCount++
      return false
    }
    return true // keep to prevent empty quarter
  })

  const clampedActions = validActions.map(action => ({
    ...action,
    favorChange: Math.max(
      -MAX_FAVOR_CHANGE_PER_ACTION,
      Math.min(MAX_FAVOR_CHANGE_PER_ACTION, action.favorChange),
    ),
  }))

  // Also filter chat messages from non-existent NPCs
  const validMessages = npcOutput.chatMessages.filter(msg =>
    activeNames.has(msg.sender),
  )

  return {
    ...npcOutput,
    npcActions: clampedActions,
    chatMessages: validMessages,
  }
}

/**
 * Validate critical period choices.
 * Rules from spec section 七:
 * - staminaCost cannot exceed remaining stamina
 * - category must be in the critical period's allowed categories
 * - stat changes must not push attributes beyond 0-100 bounds
 * - At least 1 valid choice must remain
 */
export function validateChoices(
  choices: CriticalChoice[],
  remainingStamina: number,
  criticalType: CriticalPeriodType,
  playerAttributes: PlayerAttributes,
): CriticalChoice[] {
  const allowedCategories = CRITICAL_PERIOD_CATEGORIES[criticalType] ?? []

  function clampStatChanges(
    statChanges?: CriticalChoice['effects']['statChanges'],
  ): CriticalChoice['effects']['statChanges'] {
    if (!statChanges) {
      return undefined
    }

    return Object.fromEntries(
      Object.entries(statChanges).map(([key, val]) => {
        const current = playerAttributes[key as keyof PlayerAttributes] ?? 0
        if (key === 'money') return [key, val] // money is uncapped
        const clamped = Math.max(-current, Math.min(100 - current, val as number))
        return [key, clamped]
      }),
    )
  }

  // First pass: filter invalid choices
  let valid = choices.filter(choice => {
    if (choice.staminaCost > remainingStamina) return false
    if (allowedCategories.length > 0 && !allowedCategories.includes(choice.category)) return false
    return true
  })

  // Clamp stat changes to not exceed attribute bounds (0-100, money uncapped)
  valid = valid.map(choice => ({
    ...choice,
    effects: {
      ...choice.effects,
      statChanges: clampStatChanges(choice.effects.statChanges),
    },
  }))

  // Guarantee at least 1 choice (adjust stamina/category and sanitize effects)
  if (valid.length === 0 && choices.length > 0) {
    const fallback = {
      ...choices[0],
      effects: {
        ...choices[0].effects,
        statChanges: clampStatChanges(choices[0].effects.statChanges),
      },
    }
    fallback.staminaCost = Math.min(fallback.staminaCost, remainingStamina)
    if (allowedCategories.length > 0) {
      fallback.category = allowedCategories[0]
    }
    valid = [fallback]
  }

  return valid
}

// src/ai/orchestration/quarterly.ts
import { runWorldAgent } from '@/ai/agents/world'
import { runEventAgent } from '@/ai/agents/event'
import { runNPCAgent } from '@/ai/agents/npc'
import { runNarrativeAgent } from '@/ai/agents/narrative'
import { validateEvents, validateEventNPCConsistency, validateNPCActions } from '@/ai/orchestration/conflict'
import { getRecentHistory, createQuarterSummary } from '@/ai/orchestration/history'
import { settleQuarter } from '@/engine/quarter'
import { applyStatChanges } from '@/engine/attributes'
import type { QuarterPlan } from '@/types/actions'
import type { GameState } from '@/types/game'
import type {
  AgentInput, WorldAgentOutput,
  EventAgentOutput, NPCAgentOutput,
} from '@/types/agents'

export interface QuarterlyPipelineResult {
  state: GameState
  narrative: string
  worldContext: WorldAgentOutput
  events: EventAgentOutput['events']
  npcActions: NPCAgentOutput['npcActions']
  phoneMessages: Array<{ app: string; content: string; sender?: string }>
  performanceRating?: string
  salaryChange?: number
}

export async function runQuarterlyPipeline(
  state: GameState,
  plan: QuarterPlan,
): Promise<QuarterlyPipelineResult> {
  const beforeAttributes = { ...state.player }

  // 1. Game engine: apply hard rule calculations
  const engineResult = settleQuarter(state, plan)
  const settledState = engineResult.state

  // Build agent input from SETTLED state (post-engine, so agents see updated attributes/quarter)
  const recentHistory = getRecentHistory(settledState.history)
  const agentInput: AgentInput = { state: settledState, recentHistory }

  // 2. World Agent: economy judgment
  const worldOutput = await runWorldAgent(agentInput)

  // 3. Event Agent: generate events (with world context)
  const rawEventOutput = await runEventAgent(agentInput, worldOutput)
  const worldValidatedEvents = validateEvents(rawEventOutput.events, worldOutput)
  const npcValidatedEvents = validateEventNPCConsistency(worldValidatedEvents, state.npcs)
  const eventOutput: EventAgentOutput = {
    ...rawEventOutput,
    events: npcValidatedEvents,
  }

  // 4. NPC Agent: reactions (with world + events + player actions)
  const rawNPCOutput = await runNPCAgent(
    agentInput, worldOutput, eventOutput, plan.actions,
  )
  const npcOutput = validateNPCActions(rawNPCOutput, state.npcs)

  // 5. Apply NPC favor changes to settled state
  for (const action of npcOutput.npcActions) {
    const npc = settledState.npcs.find(n => n.name === action.npcName)
    if (npc) {
      npc.favor = Math.max(0, Math.min(100, npc.favor + action.favorChange))
    }
  }

  // 6. Apply new/departed NPCs
  if (npcOutput.newNpcs) {
    settledState.npcs.push(...npcOutput.newNpcs)
  }
  if (npcOutput.departedNpcs) {
    for (const id of npcOutput.departedNpcs) {
      const npc = settledState.npcs.find(n => n.id === id)
      if (npc) npc.isActive = false
    }
  }

  // 7. Update world state
  settledState.world = {
    economyCycle: worldOutput.economy,
    industryTrends: worldOutput.trends,
    companyStatus: worldOutput.companyStatus,
  }

  // 8. Add phone messages
  const allMessages = [
    ...eventOutput.phoneMessages,
    ...npcOutput.chatMessages.map(m => ({ app: m.app, content: m.content, sender: m.sender })),
  ]
  for (const msg of allMessages) {
    settledState.phoneMessages.push({
      id: `msg_q${settledState.currentQuarter}_${Date.now()}_${Math.random().toString(36).slice(2, 6)}`,
      app: msg.app as any,
      sender: msg.sender ?? '系统',
      content: msg.content,
      read: false,
      quarter: settledState.currentQuarter,
    })
  }

  // 9. Narrative Agent: generate story
  const narrativeOutput = await runNarrativeAgent(
    agentInput, worldOutput, eventOutput, npcOutput, plan.actions, false,
  )

  // 10. Final engine validation: clamp all attributes to valid bounds
  for (const key of Object.keys(settledState.player) as Array<keyof typeof settledState.player>) {
    if (key === 'money') continue
    const val = settledState.player[key]
    settledState.player[key] = Math.max(0, Math.min(100, val))
  }
  for (const npc of settledState.npcs) {
    npc.favor = Math.max(0, Math.min(100, npc.favor))
  }

  // 11. Create quarter summary and add to history
  const eventTitles = npcValidatedEvents.map(e => e.title)
  const npcChanges = npcOutput.npcActions.map(a => `${a.npcName}好感${a.favorChange > 0 ? '+' : ''}${a.favorChange}`)
  const summary = createQuarterSummary(
    settledState.currentQuarter,
    beforeAttributes,
    settledState.player,
    eventTitles,
    npcChanges,
    narrativeOutput.narrativeSummary ?? narrativeOutput.narrative.slice(0, 100),
  )
  settledState.history.push(summary)

  return {
    state: settledState,
    narrative: narrativeOutput.narrative,
    worldContext: worldOutput,
    events: npcValidatedEvents,
    npcActions: npcOutput.npcActions,
    phoneMessages: allMessages,
    performanceRating: engineResult.performanceRating,
    salaryChange: engineResult.salaryChange,
  }
}

import { runEventAgent } from '@/ai/agents/event'
import { runNarrativeAgent } from '@/ai/agents/narrative'

import type { CriticalChoice } from '@/types/actions'
import type { GameState, CriticalPeriodType } from '@/types/game'
import type { AgentInput, WorldAgentOutput, EventAgentOutput } from '@/types/agents'

export interface CriticalDayPipelineResult {
  state: GameState
  narrative: string
  nextChoices?: CriticalChoice[]
  isComplete: boolean
  npcActions: Array<{ npcName: string; action: string; favorChange: number }>
}

function buildCriticalStoryState(
  originalState: GameState,
  settledState: GameState,
  choice: CriticalChoice,
): GameState {
  const storyState: GameState = JSON.parse(JSON.stringify(originalState))

  // Use post-choice resolved stats/NPC state, but keep the ORIGINAL critical-period day/timeMode
  // so agents narrate "today" rather than the already-advanced next day.
  storyState.player = JSON.parse(JSON.stringify(settledState.player))
  storyState.npcs = JSON.parse(JSON.stringify(settledState.npcs))
  storyState.world = JSON.parse(JSON.stringify(settledState.world))
  storyState.phoneMessages = JSON.parse(JSON.stringify(settledState.phoneMessages))
  storyState.history = JSON.parse(JSON.stringify(settledState.history))
  storyState.staminaRemaining = Math.max(0, originalState.staminaRemaining - choice.staminaCost)

  return storyState
}

export async function runCriticalDayPipeline(
  state: GameState,
  choice: CriticalChoice,
): Promise<CriticalDayPipelineResult> {
  // 1. Game engine: apply choice effects
  const engineResult = settleCriticalDay(state, choice)
  const settledState = engineResult.state

  // Build story context from "today after the choice happened, before day advancement".
  // This keeps the daily story on the correct day, while settledState remains the saved next state.
  const storyState = buildCriticalStoryState(state, settledState, choice)
  const recentHistory = getRecentHistory(storyState.history)
  const agentInput: AgentInput = { state: storyState, recentHistory }

  // Use current world state (no World Agent call during critical periods)
  const worldContext: WorldAgentOutput = {
    economy: storyState.world.economyCycle,
    trends: storyState.world.industryTrends,
    companyStatus: storyState.world.companyStatus,
    newsItems: [],
  }

  // 2. NPC Agent: dialogue and reactions to today's choice
  // Critical period choices use free-text labels, not GameAction enums,
  // so we pass the raw choice label as playerContext string
  const rawNPCOutput = await runNPCAgent(
    agentInput, worldContext,
    { events: [], phoneMessages: [] },
    [], // no ActionAllocation in critical mode
    `玩家选择了：${choice.label}（${choice.category}）`,
  )
  const npcOutput = validateNPCActions(rawNPCOutput, settledState.npcs)

  // Apply NPC favor changes
  for (const action of npcOutput.npcActions) {
    const npc = settledState.npcs.find(n => n.name === action.npcName)
    if (npc) {
      npc.favor = Math.max(0, Math.min(100, npc.favor + action.favorChange))
    }
  }

  // 3. Event Agent: check for daily complications
  const eventOutput: EventAgentOutput = await runEventAgent(agentInput, worldContext)

  // 4. Narrative Agent: daily story + next choices
  const isCriticalStill = !engineResult.isComplete
  const narrativeOutput = await runNarrativeAgent(
    agentInput,
    worldContext, eventOutput, npcOutput,
    [], // no ActionAllocation in critical mode
    true, // always use critical-day narrative mode for this day's story
    `玩家选择了：${choice.label}（${choice.category}）`,
    isCriticalStill, // only generate next-day choices if the critical period continues
  )

  // 5. Validate next day's choices if still in critical period
  let nextChoices: CriticalChoice[] | undefined
  if (isCriticalStill && narrativeOutput.choices) {
    nextChoices = validateChoices(
      narrativeOutput.choices,
      settledState.staminaRemaining,
      settledState.criticalPeriod!.type as CriticalPeriodType,
      settledState.player,
    )
  }

  return {
    state: settledState,
    narrative: narrativeOutput.narrative,
    nextChoices,
    isComplete: engineResult.isComplete,
    npcActions: npcOutput.npcActions,
  }
}
```

// src/app/api/game/turn/route.ts
import { runQuarterlyPipeline } from '@/ai/orchestration/quarterly'
import { runCriticalDayPipeline } from '@/ai/orchestration/critical'
import { AgentInputSchema } from '@/ai/schemas'
import type { CriticalChoice, QuarterPlan } from '@/types/actions'
import type { GameState } from '@/types/game'

export async function POST(request: Request) {
  try {
    const body = (await request.json()) as {
      state?: GameState
      plan?: QuarterPlan
      choice?: CriticalChoice
    }

    if (!body.state) {
      return Response.json(
        { success: false, error: 'Missing state' },
        { status: 400 },
      )
    }

    // Validate incoming state structure
    const stateCheck = AgentInputSchema.shape.state.safeParse(body.state)
    if (!stateCheck.success) {
      return Response.json(
        { success: false, error: `Invalid state: ${stateCheck.error.message}` },
        { status: 400 },
      )
    }

    const state = body.state

    if (state.timeMode === 'critical') {
      if (!body.choice) {
        return Response.json(
          { success: false, error: 'Missing choice for critical period action' },
          { status: 400 },
        )
      }

      const result = await runCriticalDayPipeline(state, body.choice)
      return Response.json({
        success: true,
        state: result.state,
        narrative: result.narrative,
        nextChoices: result.nextChoices,
        isComplete: result.isComplete,
        npcActions: result.npcActions,
      })
    }

    // Quarterly mode
    if (!body.plan) {
      return Response.json(
        { success: false, error: 'Missing plan for quarterly action' },
        { status: 400 },
      )
    }

    const result = await runQuarterlyPipeline(state, body.plan)
    return Response.json({
      success: true,
      state: result.state,
      narrative: result.narrative,
      worldContext: result.worldContext,
      events: result.events,
      npcActions: result.npcActions,
      phoneMessages: result.phoneMessages,
      performanceRating: result.performanceRating,
      salaryChange: result.salaryChange,
    })
  } catch (error) {
    const message = error instanceof Error ? error.message : 'Unknown error'
    return Response.json(
      { success: false, error: message },
      { status: 500 },
    )
  }
}
